// ─── FLOATING LIPS BACKGROUND ────────────────────────────────────────────────
interface Lip { x: number; y: number; vx: number; vy: number; depth: number; }

const FloatingLips = memo(() => {
  const containerRef = useRef<HTMLDivElement>(null);
  const rafRef = useRef<number | null>(null);
  const mouse = useRef({ x: -999, y: -999 });

  useEffect(() => {
    const handleMouseMove = (e: MouseEvent) => { mouse.current = { x: e.clientX, y: e.clientY }; };
    const handleTouchMove = (e: TouchEvent) => { const t = e.touches[0]; mouse.current = { x: t.clientX, y: t.clientY }; };
    const handleTouchEnd = () => { mouse.current = { x: -999, y: -999 }; };
    window.addEventListener("mousemove", handleMouseMove);
    window.addEventListener("touchmove", handleTouchMove, { passive: true });
    window.addEventListener("touchend", handleTouchEnd);

    const lips: Lip[] = Array.from({ length: 12 }, () => {
      const layer = Math.random();
      const depth = layer < 0.33 ? 0.5 : layer < 0.66 ? 0.85 : 1.2;
      return { x: Math.random() * window.innerWidth, y: Math.random() * window.innerHeight, vx: (Math.random() - 0.5) * 0.1 * depth, vy: (Math.random() - 0.5) * 0.1 * depth, depth };
    });

    const animate = () => {
      rafRef.current = requestAnimationFrame(animate);
      const time = Date.now() * 0.001;
      const windX = Math.cos(time * (Math.PI * 2 / 120)) * 0.002;
      const windY = Math.sin(time * (Math.PI * 2 / 120)) * 0.002;

      lips.forEach((lip, i) => {
        lip.vx += windX * lip.depth + (Math.random() - 0.5) * 0.004;
        lip.vy += windY * lip.depth + (Math.random() - 0.5) * 0.004;
        const dx = lip.x - mouse.current.x, dy = lip.y - mouse.current.y;
        const dist = Math.sqrt(dx * dx + dy * dy);
        if (dist < 160 && dist > 0) { lip.vx += dx * 0.0007; lip.vy += dy * 0.0007; }
        lip.vx *= 0.985; lip.vy *= 0.985;
        lip.x += lip.vx * 16; lip.y += lip.vy * 16;
        if (lip.x > window.innerWidth + 100) lip.x = -100;
        if (lip.x < -100) lip.x = window.innerWidth + 100;
        if (lip.y > window.innerHeight + 100) lip.y = -100;
        if (lip.y < -100) lip.y = window.innerHeight + 100;

        const el = containerRef.current?.children[i] as HTMLElement;
        if (!el) return;
        const wobbleX = Math.sin(time * 0.8 + i * 1.3) * (5 * lip.depth);
        const wobbleY = Math.cos(time * 0.6 + i * 1.7) * (5 * lip.depth);
        const rot = Math.sin(time * 0.7 + i * 1.2) * 15;
        el.style.transform = `translate(${lip.x + wobbleX}px, ${lip.y + wobbleY}px) scale(${lip.depth}) rotate(${rot}deg)`;
        el.style.opacity = `${0.08 + lip.depth * 0.12}`;
      });
    };
    animate();

    return () => {
      window.removeEventListener("mousemove", handleMouseMove);
      window.removeEventListener("touchmove", handleTouchMove);
      window.removeEventListener("touchend", handleTouchEnd);
      if (rafRef.current) cancelAnimationFrame(rafRef.current);
    };
  }, []);

  const staticShuffle = [1, 5, 3, 7, 2, 6, 4, 1, 5, 3, 7, 2];
  return (
    <div ref={containerRef} className="fixed inset-0 pointer-events-none z-0 overflow-hidden">
      {staticShuffle.map((img, i) => (
        <div key={i} className="absolute" style={{ width: 80, height: 80, willChange: "transform" }}>
          <Image src={`/lips/${img}.png`} alt="" width={80} height={80} className="object-contain w-full h-full" />
        </div>
      ))}
    </div>
  );
});
FloatingLips.displayName = "FloatingLips";
